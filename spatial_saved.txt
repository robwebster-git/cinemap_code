DECLARE
  busroute_geom SDO_GEOMETRY;
  cinema_geom SDO_GEOMETRY;
  dist NUMBER;
  geoma SDO_GEOMETRY;
  geomb SDO_GEOMETRY;

BEGIN
SELECT a.ora_geometry into busroute_geom FROM busroutes_8307 a;
SELECT b.geom into cinema_geom FROM s1434165.cinemas b;

SDO_GEOM.SDO_CLOSEST_POINTS(busroute_geom, cinema_geom, 0.005, NULL, dist, geoma, geomb);

DBMS_OUTPUT.PUT_LINE('dist output parameter value = ' || dist);
END;
